void main(){

	int count=0,i,j;

	for(i=0;i<5;i++){
		for(j=0;j<5;j++){
			count++;
			a[i][j] = count;
		}
	}
	for(i=0;i<5;i++){
		for(j=0;j<5;j++){
			printf("%-2d ",a[i][j]);	
		}
		putchar('\n');
	}
	//1. 각행의 총합 출력
	int sum=0;
	for(i=0;i<5;i++){
		for(j=0;j<5;j++){
			sum += a[i][j];
		}
		printf("%d\n", sum);//각행의 총합 출력
		sum = 0;

	}
	//2. 각행의 짝수의 개수 출력
	count = 0;

	for(i=0;i<5;i++){
		for(j=0;j<5;j++){
			if(a[i][j]%2==0)
				count++;
		}
		printf("\n%d행의 짝수 개수 : %d\n",i,count);
		count = 0;
	}
	//3. 각행의 짝수의 총합 출력
	sum=0;
	for(i=0;i<5;i++){
		for(j=0;j<5;j++){
			if(a[i][j]%2==0)
				sum += a[i][j];
		}
		printf("\n%d행의 짝수의 총합 : %d\n",i,sum);
		sum = 0;
	}
	//4. 각행의 짝수의 평균
	sum=0;
	count = 0;
	for(i=0;i<5;i++){
		for(j=0;j<5;j++){
			if(a[i][j]%2==0){
				sum += a[i][j];
				count++;
			}
		}
		printf("\n%d행의 짝수의 평균 : %d\n",i,sum/count);
		sum = 0;
		count = 0;
	}


}

		switch(score[i]/10){
		case 10:
		case 9:
			grade[i] = 'A';
			break;
		case 8:
			grade[i] = 'B';
			break;
		case 7:
			grade[i] = 'C';
			break;
		case 6:
			grade[i] = 'D';

			grade[i] = 'F';

	printf("%10s %5s %5s\n","과목명","점수","등급");

}

	return a%b;
}
int max(int a,int b){
	if(a>b)
		return a;
	else
		return b;
}
int min(int a,int b){
	/*
	if(a<b)
		return a;
	else
		return b;
		*/

}

	/*
	if(a < 0)
		return -a;
	return a;
	*/

#include <stdio.h>
void r_area(double *r, double *c_area);
void v_area(double r, double c_area);
void main(){
	double r = 3.0;
	double area = 0.0;
	
	printf("Call By Value 수행전\n");
	printf("r = %lf \t area = %lf\n",r,area);
	v_area(r,area);
	printf("Call By Value 수행후\n");
	printf("r = %lf \t area = %lf\n",r,area);

	printf("Call By Refference 수행전\n");
	printf("r = %lf \t area = %lf\n",r,area);
	r_area(&r,&area);
	printf("Call By Refference 수행후\n");
	printf("r = %lf \t area = %lf\n",r,area);
	

}
void r_area(double *r, double *c_area){
	*r = *r +1.0;
	*c_area = (*r) * (*r) * 3.14;
	printf("r_area 연산 결과\n");
	printf("r = %lf \t area = %lf\n",*r,*c_area);
}
void v_area(double r, double c_area){
	r = r + 1.0;
	c_area = r * r * 3.14;
	printf("v_area 연산 결과\n");
	printf("r = %lf \t area = %lf\n",r,c_area);

}
